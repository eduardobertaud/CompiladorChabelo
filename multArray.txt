program UNO;

main(){
	var int arr1[5];
	var int arr2[5];
	var int contador;
	var int contador2;
	contador = 0;
	var int suma;
	suma = 0;
	while(contador < 5){
		arr1[contador] = contador * 1;
		arr2[contador] = contador * 2;
		contador = contador + 1;
	}
	contador2 = 0;
	while(contador2 < 5){
		suma = suma + arr1[contador2]* arr2[contador2];
		contador2 = contador2 + 1;
	}
	print(suma);
}
end
